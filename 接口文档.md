### Add a user (home / business)

``` js
{
  url: '/user'
  method: 'post',
  param: {
    username, // string [0, 15]
    password, // string [0, 20]
    role_id, // int 0-admin 1-home 2-business
    ...otherParam
  },
  return_ok: {
    code: 201,
    msg: 'User Created!'
  },
  return_fail: {
    code: errCode.USER_FAILURE,
    msg: 'User Failure'
  }
}
// if role_id is 1, extra params needed below
{
  marriage_status, // int 0-single 1-married 2-divorced
  gender, // int 0-male 1-female 2-NA
  age, // int [0, 200]
  annual_income, // bigint, 1 rmb = 100, last two digits are decimals
  street, // string [0,30]
  city, // string [0,20]
  state, // string provided by server end, see api get '/states'
  zip_code, // int length = 5
}
// if role_id is 2, extra params needed below
{
  annual_income, // bigint, 1 rmb = 100, last two digits are decimals
  street, // string [0,30]
  city, // string [0,20]
  state, // string provided by server end, see api get '/states'
  zip_code, // int length = 5
  cate, // string provided by server end, see api get 'businessCate'
}
```
